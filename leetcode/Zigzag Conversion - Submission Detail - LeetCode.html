class Solution(object):
def convert(self, s, numRows):
"""
:type s: str
:type numRows: int
:rtype: str
"""
word = []
ind = 0
# offset alternates for calculating indices
off = 0
# iterate through rows
for i in range(0,numRows):
off = 0
if (i+1 > len(s)):
return "".join(word)
word.append(s[ind])
if (i == 0 or i == numRows-1 and len(s) > 1):
if (numRows == 1):
if (len(s) > 1):
ind += 1
while (ind < len(s)):
word.append(s[ind])
ind += 1
else:
while (ind+(2*numRows)-2 < len(s)):
ind += 2*numRows-2
word.append(s[ind])
else:
while ((off == 0 and ind+(2*numRows)-(2*i)-2 < len(s)) or
(off == 1 and ind+2*i < len(s))):
if (off == 0):
if (ind+(2*numRows)-(2*i)-2 < len(s)):
ind += 2*numRows - 2*i - 2
word.append(s[ind])
off = 1
continue
if (off == 1):
if (ind + 2*i < len(s)):
ind += 2*i
word.append(s[ind])
off = 0
continue
ind = i + 1
the_word = "".join(word)
print(word)
return the_word
